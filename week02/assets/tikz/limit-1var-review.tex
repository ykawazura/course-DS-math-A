\begin{tikzpicture}
  % Axes
  \draw[axis] (-1,0) -- (5.5,0) node[below] {$x$};
  \draw[axis] (0,-0.5) -- (0,4.5) node[left] {$f(x)$};

  % Curve: f(x) = 0.04x^3 - 0.3x^2 + 1.1x + 1 (monotonically increasing)
  \draw[very thick, domain=0.3:5, samples=80]
    plot (\x, {0.04*\x*\x*\x - 0.3*\x*\x + 1.1*\x + 1});

  % Limit point
  \pgfmathsetmacro{\xa}{2.5}
  \pgfmathsetmacro{\ya}{0.04*\xa*\xa*\xa - 0.3*\xa*\xa + 1.1*\xa + 1}

  % Dashed lines to axes
  \draw[dashed, semithick] (\xa,0) -- (\xa,\ya);
  \draw[dashed, semithick] (0,\ya) -- (\xa,\ya);

  % Open dot at limit point
  \node[open dot] at (\xa,\ya) {};

  % Labels
  \node[below] at (\xa,0) {$a$};
  \node[left] at (0,\ya) {$A$};

  % Left-side approach (red arrow)
  \draw[red, very thick, -{Stealth[length=6pt]}, domain=0.8:2.2, samples=30]
    plot (\x, {0.04*\x*\x*\x - 0.3*\x*\x + 1.1*\x + 1});
  \node[red, below, font=\small] at (1.3,1.7) {左側極限};

  % Right-side approach (blue arrow)
  \draw[blue, very thick, -{Stealth[length=6pt]}, domain=4.5:2.8, samples=30]
    plot (\x, {0.04*\x*\x*\x - 0.3*\x*\x + 1.1*\x + 1});
  \node[blue, above, font=\small] at (4.0,3.5) {右側極限};
