\begin{tikzpicture}
  % Axes
  \draw[axis] (-2.5,0) -- (4.5,0) node[below] {$x$};
  \draw[axis] (0,-0.5) -- (0,7) node[left] {$y$};

  % Curve: y = x^2 + 1
  \draw[very thick, domain=-2.2:3.2, samples=60] plot (\x, {0.7*\x*\x + 1});

  % Point (2, 5)
  \pgfmathsetmacro{\xa}{2}
  \pgfmathsetmacro{\ya}{0.7*\xa*\xa + 1}
  \fill[red] (\xa,\ya) circle (3pt);

  % Dashed lines to axes
  \draw[dashed, semithick] (\xa,0) -- (\xa,\ya);
  \draw[dashed, semithick] (0,\ya) -- (\xa,\ya);

  % Labels
  \node[below] at (\xa,0) {$2$};
  \node[left] at (0,\ya) {$5$};

  % Red arrows approaching along curve
  \draw[red, very thick, -{Stealth[length=6pt]}, domain=0.5:1.7, samples=30]
    plot (\x, {0.7*\x*\x + 1});
  \draw[red, very thick, -{Stealth[length=6pt]}, domain=3.0:2.3, samples=30]
    plot (\x, {0.7*\x*\x + 1});
\end{tikzpicture}
