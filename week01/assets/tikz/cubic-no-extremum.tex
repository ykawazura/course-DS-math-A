\begin{tikzpicture}
  % f(x) = x^3
  \begin{scope}
    \draw[axis] (-2.2,0) -- (2.2,0) node[below] {$x$};
    \draw[axis] (0,-2.5) -- (0,2.5) node[left] {$f(x)$};
    \draw[very thick, domain=-1.35:1.35, samples=50] plot (\x, {\x*\x*\x});
    \fill (0,0) circle (3pt);
    \node[right, font=\small] at (1.2,2) {$f(x) = x^3$};
  \end{scope}

  % f'(x) = 3x^2
  \begin{scope}[xshift=6.5cm]
    \draw[axis] (-2.2,0) -- (2.2,0) node[below] {$x$};
    \draw[axis] (0,-0.5) -- (0,3.5) node[left] {$f'(x)$};
    \draw[very thick, domain=-1.4:1.4, samples=50] plot (\x, {1.5*\x*\x});
    \fill (0,0) circle (3pt);
    \node[right, font=\small] at (1.5,3.2) {$f'(x) = 3x^2$};
    \node[below left, font=\small] at (0,0) {$f'(0)=0$};
  \end{scope}
\end{tikzpicture}
