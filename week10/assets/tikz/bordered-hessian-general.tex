\documentclass[tikz,border=5pt]{standalone}
\usepackage{luatexja}
\usepackage[hiragino-pron]{luatexja-preset}
\usepackage{amsmath,amssymb}
\usetikzlibrary{matrix,decorations.pathreplacing,positioning,fit}

\begin{document}
\begin{tikzpicture}[
  every node/.style={font=\large},
  brace/.style={decorate, decoration={brace, amplitude=5pt}},
  mbrace/.style={decorate, decoration={brace, amplitude=5pt, mirror}},
]

% === First matrix (L-subscript form) ===
\matrix[matrix of math nodes,
  row sep=6pt, column sep=10pt] (A) {
  \mathcal{L}_{\lambda_1\lambda_1} & \cdots & \mathcal{L}_{\lambda_1\lambda_m}
    & \mathcal{L}_{\lambda_1 x_1} & \cdots & \mathcal{L}_{\lambda_1 x_n} \\
  \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
  \mathcal{L}_{\lambda_m\lambda_1} & \cdots & \mathcal{L}_{\lambda_m\lambda_m}
    & \mathcal{L}_{\lambda_m x_1} & \cdots & \mathcal{L}_{\lambda_m x_n} \\
  \mathcal{L}_{x_1\lambda_1} & \cdots & \mathcal{L}_{x_1\lambda_m}
    & \mathcal{L}_{x_1 x_1} & \cdots & \mathcal{L}_{x_1 x_n} \\
  \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
  \mathcal{L}_{x_n\lambda_1} & \cdots & \mathcal{L}_{x_n\lambda_m}
    & \mathcal{L}_{x_n x_1} & \cdots & \mathcal{L}_{x_n x_n} \\
};

% Left braces for row groups
\draw[mbrace, raise=3pt]
  ([xshift=-3pt]A-1-1.north west) -- ([xshift=-3pt]A-3-1.south west)
  node[midway, left=7pt] (Am) {$m$};
\draw[mbrace, raise=3pt]
  ([xshift=-3pt]A-4-1.north west) -- ([xshift=-3pt]A-6-1.south west)
  node[midway, left=7pt] (An) {$n$};

% Parentheses via fit node (encompasses braces + labels)
\node[fit=(A)(Am)(An), left delimiter=(, right delimiter=),
  inner xsep=3pt, inner ysep=1pt] (Adelim) {};

% H bar = label
\node[left=10pt] at (Adelim.west) {$\bar{H} =$};

% Overbraces for columns
\draw[brace, raise=4pt]
  (A-1-1.north west) -- (A-1-3.north east) node[midway, above=9pt] {$m$};
\draw[brace, raise=4pt]
  (A-1-4.north west) -- (A-1-6.north east) node[midway, above=9pt] {$n$};

% === Second matrix (explicit form) ===
\matrix[matrix of math nodes,
  row sep=6pt, column sep=10pt, anchor=north] (B) at ([yshift=-30pt]A.south) {
  0 & \cdots & 0
    & -g_{1,x_1} & \cdots & -g_{1,x_n} \\
  \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
  0 & \cdots & 0
    & -g_{m,x_1} & \cdots & -g_{m,x_n} \\
  -g_{1,x_1} & \cdots & -g_{m,x_1}
    & \mathcal{L}_{x_1 x_1} & \cdots & \mathcal{L}_{x_1 x_n} \\
  \vdots & \ddots & \vdots & \vdots & \ddots & \vdots \\
  -g_{1,x_n} & \cdots & -g_{m,x_n}
    & \mathcal{L}_{x_n x_1} & \cdots & \mathcal{L}_{x_n x_n} \\
};

% Left braces
\draw[mbrace, raise=3pt]
  ([xshift=-3pt]B-1-1.north west) -- ([xshift=-3pt]B-3-1.south west)
  node[midway, left=7pt] (Bm) {$m$};
\draw[mbrace, raise=3pt]
  ([xshift=-3pt]B-4-1.north west) -- ([xshift=-3pt]B-6-1.south west)
  node[midway, left=7pt] (Bn) {$n$};

% Parentheses via fit node
\node[fit=(B)(Bm)(Bn), left delimiter=(, right delimiter=),
  inner xsep=3pt, inner ysep=1pt] (Bdelim) {};

% = sign
\node[left=10pt] at (Bdelim.west) {$=$};

% Overbraces
\draw[brace, raise=4pt]
  (B-1-1.north west) -- (B-1-3.north east) node[midway, above=9pt] {$m$};
\draw[brace, raise=4pt]
  (B-1-4.north west) -- (B-1-6.north east) node[midway, above=9pt] {$n$};

\end{tikzpicture}
\end{document}
